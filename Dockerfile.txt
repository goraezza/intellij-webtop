# Imagen base con soporte gráfico (webtop) y Ubuntu
FROM ghcr.io/linuxserver/webtop:ubuntu-xfce

# Instala dependencias necesarias para AppImage
USER root
RUN apt update && apt install -y \
    libfuse2 \
    wget \
    tar \
    xz-utils \
    libxss1 \
    libatk-bridge2.0-0 \
    libgtk-3-0 \
    libasound2 \
    libnss3 \
    libx11-xcb1 \
    libxcb-dri3-0 \
    --no-install-recommends && \
    apt clean && rm -rf /var/lib/apt/lists/*

# Crea carpeta para toolbox
RUN mkdir -p /opt/toolbox

# Descarga JetBrains Toolbox como AppImage
RUN wget -O /opt/toolbox/jetbrains-toolbox.AppImage https://download.jetbrains.com/toolbox/jetbrains-toolbox-2.3.1.11116.AppImage && \
    chmod +x /opt/toolbox/jetbrains-toolbox.AppImage && \
    /opt/toolbox/jetbrains-toolbox.AppImage --appimage-extract && \
    mv squashfs-root /opt/jetbrains-toolbox && \
    ln -s /opt/jetbrains-toolbox/AppRun /usr/local/bin/jetbrains-toolbox

# Cambia al usuario por defecto de Webtop
USER abc

# Ejecutar Webtop normalmente. IntelliJ se abrirá desde el menú de escritorio dentro del contenedor
